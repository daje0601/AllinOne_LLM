  Google Cloud SDK 설치 및 설정 순서

  # 1. 설치되어 있는지 확인
  which gcloud

  # 2. Google Cloud SDK 설치
  curl https://sdk.cloud.google.com | bash

  # 3. 환경 파일 로드
  source ~/google-cloud-sdk/path.bash.inc && source ~/google-cloud-sdk/completion.bash.inc

  # 4. PATH에 추가하고 버전 확인
  export PATH=$PATH:~/google-cloud-sdk/bin && gcloud --version

  # 5. bashrc에 영구 추가 (재부팅 후에도 유지)
  echo 'export PATH=$PATH:~/google-cloud-sdk/bin' >> ~/.bashrc && echo 'source ~/google-cloud-sdk/completion.bash.inc' >> ~/.bashrc
